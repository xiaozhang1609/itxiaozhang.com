class ProxyPool:
    """ä»£ç†IPæ± ç®¡ç†ç±»"""
    def __init__(self):

    def add_proxy(self, proxy):
        """æ·»åŠ ä»£ç†"""
        self.proxies.put(proxy)

    def get_proxy(self):
        """è·å–ä»£ç†"""
        try:
            return self.proxies.get()

            return None

    def remove_proxy(self, proxy):
        """ç§»é™¤å¤±æ•ˆä»£ç†"""

            if proxy in self.proxies.queue:
                self.proxies.queue.remove(proxy)
